void fetchTheStars()
{
    repeat (10) {
        turnLeft();
        go();
        dropBeeper();
        turnLeft();
        if (frontIsClear()) {
            moveForward();
        }
    }
}

void go() {
    if (!beeperAhead()) {
        moveForward();
        go();
    }
    else {
        moveForward();
        pickBeeper();
        turnAround();
    }
    moveForward();
}
